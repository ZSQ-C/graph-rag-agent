```python
    """整合的缓存检查方法，优化性能"""

        self.global_cache_manager.set(query, cached_response)
```

```python
    def __init__(self, cache_dir: str = "./cache/fusion_graphrag") -> None:
        self.multi_agent = MultiAgentFacade()  # 多智能体门面
        self._global_cache: Dict[str, str] = {}

        cached = self._read_cache(query, thread_id)

        # 缓存结果
        self._write_cache(query, thread_id, answer)
```

```python
def get(self, query: str, skip_validation: bool = False, thread_id: str = "default", **kwargs) -> Optional[str]:
    """获取缓存内容，支持跳过验证优化性能"""
    
    # 生成缓存键
    cache_key = self.key_strategy.generate_key(query, thread_id=thread_id, **kwargs)
    
    # 精确匹配
    cached_item = self.storage.get(cache_key)
    if cached_item and cached_item.is_valid():
        if skip_validation or self._validate_cached_item(cached_item, query):
            self.performance_metrics['exact_hits'] += 1
            return cached_item.content
    
    # 向量相似性匹配
    if self.enable_vector_similarity and self.vector_matcher:
        similar_item = self.vector_matcher.find_similar(query, self.storage)
        if similar_item and similar_item.is_valid():
            if skip_validation or self._validate_cached_item(similar_item, query):
                self.performance_metrics['vector_hits'] += 1
                return similar_item.content
    
    self.performance_metrics['misses'] += 1
    return None
```

```python
# 上下文感知键策略
class ContextAwareCacheKeyStrategy(CacheKeyStrategy):
    def generate_key(self, query: str, thread_id: str = "default", **kwargs) -> str:
        low_level_keywords = kwargs.get('low_level_keywords', [])
        high_level_keywords = kwargs.get('high_level_keywords', [])
        
        key_parts = [
            thread_id,
            query.strip().lower(),
            "|".join(sorted(low_level_keywords)),
            "|".join(sorted(high_level_keywords))
        ]
        return "__".join(key_parts)
```

```python
class TaskGraph(BaseModel):
    """任务依赖图 - 轻量级DAG实现"""
    
    nodes: List[TaskNode]
    execution_mode: Literal["sequential", "parallel", "adaptive"]
    
    def topological_sort(self) -> List[TaskNode]:
        """拓扑排序算法实现"""
        task_map = {node.task_id: node for node in self.nodes}
        in_degree: Dict[str, int] = {node.task_id: 0 for node in self.nodes}
        adjacency: Dict[str, List[str]] = defaultdict(list)
        
        # 构建邻接表
        for node in self.nodes:
            for dep_id in node.depends_on:
                adjacency[dep_id].append(node.task_id)
                in_degree[node.task_id] += 1
        
        # Kahn 算法拓扑排序
        queue = deque(sorted(
            (task_map[task_id] for task_id, degree in in_degree.items() if degree == 0),
            key=lambda x: (x.priority, x.task_id),
        ))
        
        ordered_nodes: List[TaskNode] = []
        while queue:
            current = queue.popleft()
            ordered_nodes.append(current)
            
            for neighbor_id in adjacency.get(current.task_id, []):
                in_degree[neighbor_id] -= 1
                if in_degree[neighbor_id] == 0:
                    queue.append(task_map[neighbor_id])
            
            queue = deque(sorted(list(queue), key=lambda x: (x.priority, x.task_id)))
        
        return ordered_nodes
```

```python
class BaseEvaluator(ABC):
    def __init__(self, config):
        # 动态收集所有评估指标类
        self.available_metrics = self._collect_metrics()
        
        # 初始化配置的评估指标
        self.metric_class = {}
        for metric in self.metrics:
            if metric in self.available_metrics:
                self.metric_class[metric] = self.available_metrics[metric](self.config.to_dict())
            else:
                raise NotImplementedError(f"评估指标 {metric} 未实现")
    
    def _collect_metrics(self) -> Dict[str, Type[BaseMetric]]:
        """递归收集所有继承自BaseMetric的评估指标类"""
        
        def find_descendants(base_class, subclasses=None):
            if subclasses is None:
                subclasses = set()
            
            direct_subclasses = base_class.__subclasses__()
            for subclass in direct_subclasses:
                if subclass not in subclasses:
                    subclasses.add(subclass)
                    find_descendants(subclass, subclasses)
            return subclasses
        
        available_metrics = {}
        for cls in find_descendants(BaseMetric):
            metric_name = cls.metric_name
            available_metrics[metric_name] = cls
        
        return available_metrics
```

## 关键算法实现分析

### 1. 缓存向量相似性匹配算法

**算法原理**:
- 使用 Sentence-BERT 模型计算查询文本的嵌入向量
- 构建向量索引（FAISS）进行快速相似性搜索
- 设置相似度阈值，返回最相似的缓存项

**实现细节**:
```python
class VectorSimilarityMatcher:
    def find_similar(self, query: str, storage: CacheStorageBackend) -> Optional[CacheItem]:
        """查找语义相似的缓存项"""
        
        # 计算查询向量
        query_embedding = self.embedding_provider.get_embedding(query)
        
        # 在向量索引中搜索
        if self.index and len(self.index) > 0:
            distances, indices = self.index.search(
                np.array([query_embedding]), 
                k=min(5, len(self.index))
            )
            
            # 检查相似度阈值
            for i, distance in enumerate(distances[0]):
                similarity = 1 - distance  # 转换为相似度
                if similarity >= self.similarity_threshold:
                    item_id = self.vector_ids[indices[0][i]]
                    similar_item = storage.get_by_id(item_id)
                    if similar_item and similar_item.is_valid():
                        return similar_item
        
        return None
```

### 2. 任务依赖图拓扑排序算法

**算法选择**: Kahn 算法（广度优先搜索）

**时间复杂度**: O(V + E)，其中 V 是任务数，E 是依赖边数

**实现优化**:
- 按优先级排序：高优先级任务优先执行
- 并行执行支持：无依赖任务可以并行执行
- 循环依赖检测：确保 DAG 的合法性

### 3. 社区检测权重算法

**算法原理**:
- 基于社区内实体数量计算社区权重
- 权重 = 社区内唯一实体数量
- 用于搜索结果的排序和优化

**Neo4j Cypher 实现**:
```cypher
MATCH (n:`__Community__`)<-[:IN_COMMUNITY]-()<-[:MENTIONS]-(c)
WITH n, count(distinct c) AS chunkCount
SET n.weight = chunkCount
```
